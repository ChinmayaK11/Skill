#include <stdio.h>
#include <string.h>
int longestPrefixSuffix(char *s) {
    int n = strlen(s);
    int lps[n]; // Array to store the LPS values
    int len = 0;
    int i = 1;
    
    lps[0] = 0; // First value is always 0
    
    while (i < n) {
        if (s[i] == s[len]) {
            len++;
            lps[i] = len;
            i++;
        } else {
            if (len != 0) {
                len = lps[len - 1];
            } else {
                lps[i] = 0;
                i++;
            }
        }
    }
    
    return lps[n - 1]; // Last value in LPS array gives the result
}

int main() {
    char s[100];
    printf("Enter the string: ");
    scanf("%s", s);
    printf("Length of longest prefix which is also a suffix: %d\n", longestPrefixSuffix(s));
    return 0;
}
